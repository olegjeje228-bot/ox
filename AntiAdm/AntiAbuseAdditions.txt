// Лимиты длины команд (можно вынести в Config.cs)
private const int MaxCommandLength = 4000;
private const int MaxCinfoLength = 1000;
private const int MaxCkeycardLength = 1000;

// Спам повторяющимися командами (одинаковый текст подряд)
private readonly Dictionary<string, (string LastQuery, DateTime LastTime, int Count)> _repeatHistory
    = new Dictionary<string, (string, DateTime, int)>();

private const int MaxRepeatCount = 5;
private const float RepeatWindowSeconds = 10f;

// Команды, которые не проверяются на повтор (админы жмут по бинду)
private static readonly HashSet<string> RepeatCheckExempt = new(StringComparer.OrdinalIgnoreCase)
{
    "noclip", "goto", "tpx", "heal",
};

// Общий rate-limit: не больше N команд за окно с одного senderId
private readonly Dictionary<string, (List<DateTime> Times, int DenyCount)> _rateLimitHistory
    = new Dictionary<string, (List<DateTime>, int)>();

private const int MaxCommandsPerWindow = 20;
private const float RateLimitWindowSeconds = 10f;

// Эскалация: N отказов за минуту -> блокировка RA
private const int MaxDenyPerMinute = 10;
private const float DenyBlockDurationSeconds = 120f;
private readonly Dictionary<string, DateTime> _denyBlockedUntil = new Dictionary<string, DateTime>();

// AA-22: общая длина команды

{
    Deny(ev, "AA-22");

// Проверка блокировки по эскалации
if (_denyBlockedUntil.TryGetValue(senderId, out var blockedUntil) && DateTime.UtcNow < blockedUntil)
{
    Deny(ev, "AA-27");
    return;
}

// AA-28: общий rate-limit (любые команды, не только одинаковые)
if (!_rateLimitHistory.TryGetValue(senderId, out var rl))
{
    rl = (new List<DateTime>(), 0);
    _rateLimitHistory[senderId] = rl;
}
var cutoff = DateTime.UtcNow.AddSeconds(-RateLimitWindowSeconds);
rl.Times.RemoveAll(t => t < cutoff);
rl.Times.Add(DateTime.UtcNow);
if (rl.Times.Count > MaxCommandsPerWindow)
{
    Deny(ev, "AA-28");
    return;
}

// AA-23: защита от спама одинаковой командой (только для тяжёлых, не для exempt)
if (!RepeatCheckExempt.Contains(cmd))
{
    // Ленивое приведение к нижнему регистру — только если команда не в whitelist
    string queryLower = ev.Query.ToLowerInvariant();
    if (_repeatHistory.TryGetValue(senderId, out var last))
    {
        bool sameQuery = string.Equals(last.LastQuery, queryLower, StringComparison.Ordinal);
        bool withinWindow = (DateTime.UtcNow - last.LastTime).TotalSeconds < RepeatWindowSeconds;

        if (sameQuery && withinWindow)
        {
            int newCount = last.Count + 1;
            _repeatHistory[senderId] = (queryLower, DateTime.UtcNow, newCount);
            if (newCount > MaxRepeatCount)
            {
                Deny(ev, "AA-23");
                return;
            }
        }
        else
        {
            _repeatHistory[senderId] = (queryLower, DateTime.UtcNow, 1);
        }
    }
    else
    {
        _repeatHistory[senderId] = (queryLower, DateTime.UtcNow, 1);
    }
}

    }
}

    Deny(ev, "AA-31");
    return;

// --- 7. Кулдаун на команды, влияющие на раунд ---
if (cmd == "roundrestart" || cmd == "lobbylock" || cmd == "forcestart"
    || cmd == "cassie" || cmd == "announce")
{
    if (_lastCommandTime.TryGetValue(senderId, out var lastRoundCmd)
        && (DateTime.UtcNow - lastRoundCmd).TotalSeconds < 30)
    {
        Deny(ev, "AA-32");
        return;
    }
    _lastCommandTime[senderId] = DateTime.UtcNow;
}

// --- 8. Лимиты на scale/size/gravity ---

{
    if (TryParseScaleArgs(args, out float maxVal, out bool isDummy))
    {
        float limit = isDummy ? _config.AntiAdmDummyMaxScale : _config.AntiAdmMaxScale;
        if (maxVal > limit)
        {
            Deny(ev, isDummy ? "AA-03" : "AA-06");
            return;
        }
        // Дополнительно: запретить отрицательный scale
        if (maxVal <= 0f)
        {
            Deny(ev, "AA-33");
            return;
        }
    }
}

/// <summary>
/// Проверяет, содержит ли текст запрещённые rich-text теги TMP.
/// Разрешены только: <color>, <b>, </b>, <i>, </i>, </color>.
/// </summary>
private static bool ContainsDangerousRichText(string text)
{
    if (string.IsNullOrEmpty(text))
        return false;

    int idx = 0;
    while (true)
    {
        int open = text.IndexOf('<', idx);
        if (open < 0)
            break;

        int close = text.IndexOf('>', open);
        if (close < 0)
            break;

        string tag = text.Substring(open, close - open + 1);

        // Проверяем, разрешён ли тег
        bool allowed = false;
        if (tag.StartsWith("<color") || tag == "</color>"
            || tag == "<b>" || tag == "</b>"
            || tag == "<i>" || tag == "</i>")
        {
            allowed = true;
        }

        if (!allowed)
            return true;

        idx = close + 1;
    }

    return false;
}

/// <summary>
/// Проверяет, содержит ли текст невидимые символы Юникода.
/// </summary>
private static bool ContainsInvisibleUnicode(string text)
{
    if (string.IsNullOrEmpty(text))
        return false;

    foreach (char c in text)
    {
        // Zero-width space, zero-width non-joiner, zero-width joiner, BOM
        if (c == '\u200B' || c == '\u200C' || c == '\u200D' || c == '\uFEFF')
            return true;

        // RTL override, LTR override, pop directional formatting
        if (c == '\u202E' || c == '\u202D' || c == '\u202C')
            return true;

        // Управляющие символы (кроме разрешённых \n, \r, \t)
        if (char.GetUnicodeCategory(c) == System.Globalization.UnicodeCategory.Control
            && c != '\n' && c != '\r' && c != '\t')
            return true;
    }

    return false;
}

public void CleanupPlayer(string userId)
{
    _repeatHistory.Remove(userId);
    _rateLimitHistory.Remove(userId);
    _denyBlockedUntil.Remove(userId);
}

// --- 12. Эскалация при отказах ---
// Модифицировать метод Deny:
private void Deny(SendingValidCommandEventArgs ev, string id)
{
    ev.IsAllowed = false;
    ev.Response = $"<color=red>Отказ [{id}]</color>";

    // Эскалация
    if (_rateLimitHistory.TryGetValue(ev.Player?.UserId ?? "console", out var rl))
    {
        rl.DenyCount++;
        if (rl.DenyCount >= MaxDenyPerMinute)
        {
            string uid = ev.Player?.UserId ?? "console";
            _denyBlockedUntil[uid] = DateTime.UtcNow.AddSeconds(DenyBlockDurationSeconds);
            Log.Warn($"[AntiAdm] {uid} заблокирован на {DenyBlockDurationSeconds}с (превышен лимит отказов)");

            // Опционально: уведомление в Discord
            // DiscordWebhookService.Send($"[ANTI-ABUSE] {uid} заблокирован на {DenyBlockDurationSeconds}с ({rl.DenyCount} отказов)");
        }
    }
}
